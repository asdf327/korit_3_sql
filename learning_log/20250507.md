```py

import RPi.GPIO as g
from time import sleep

ledPin = (16,20,21)

g.setmode(g.BCM)

for pin in ledPin:
  g.setup(ledPin, g.OUT)

try:
  isOnAll = False
  while True:
    for pin is ledPin:
      if isOnAll:
      g.output(pin, g.HIGH) 
    else:
      g.output(pin, g.LOW)  
    sleep(0.1)
  isOnAll = not isOnAll
except KeyboardInterrupt:
  pass
finally:
  g.cleanup()
```
